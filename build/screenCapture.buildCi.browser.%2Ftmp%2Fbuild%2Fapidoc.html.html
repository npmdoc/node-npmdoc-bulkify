function bulkify (file, opts)

  • description and source-code
    bulkify = function (file, opts) {
    if (/\.json$/.test(file)) return through();
    if (!opts) opts = {};
    var filedir = path.dirname(file);
    var vars = opts.vars || {
        __filename: file,
        __dirname: filedir
    };

    var sm = staticModule(
        { 'bulk-require': bulkRequire },
        { vars: vars, varModules: { path: path } }
    );
    return through(function (buf, enc, next) {
        sm.write(buf)
        sm.end()
        sm.pipe(concat(function (output) {
            next(null, output)
        }))
    })

    function bulkRequire (dir, globs, opts) {
        var stream = through();
        var res = bulk(dir, globs, {
            index: opts && opts.index,
            require: function (x) {
                if (!file) return path.resolve(x);
                var r = path.relative(filedir, x);
                return /^\./.test(r) ? r : './' + r;
            }
        });
        stream.push(walk(res, opts));
        stream.push(null);
        return stream;
    }
}
